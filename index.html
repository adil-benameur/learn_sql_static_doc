-- Création d'une base de données
    CREATE DATABASE notreDB;
    -- Suppression d'une base de données
    DROP DATABASE notreDB;
    
    -- Liste les bases de données disponible
    SHOW DATABASES;
    
    -- Sélectionne une base de données
    USE notreDB;
    
    -- Retourne toutes les colonnes d'une table de notre base de données
    SELECT * FROM eleves;
    
    -- Retourne les colonnes nom et prenom
    SELECT nom,
           prenom FROM eleves;
    
    -- Crée une base de données eleve
    CREATE TABLE eleves (nom VARCHAR(20), prenom VARCHAR(20));
    
    
    -- Insère dans la table eleves le nom 'Mutt' et le prenom 'John'
    INSERT INTO eleves VALUES('Mutt','Richard');
    
    -- Mets à jour le prenom des rangés ayant pour nom 'Mutt'
    UPDATE eleves SET prenom='John' WHERE nom='Mutt';
    
    -- Supprime les rangés ayant pour nom 'Mutt'
    DELETE FROM eleves WHERE nom='Mutt';
    
    -- Suprimme toutes les rangés de la table eleves
    DELETE FROM eleves;
    
    -- Supprimme la table eleve
    DROP TABLE eleves;
